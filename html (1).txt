    <button id="next-button">Следующий вопрос</button>
    <script>
      // Обработчик клика по кнопке "Следующий вопрос"
      document.getElementById("next-button").addEventListener("click", function() {
        // Получаем все выбранные ответы на текущий вопрос
        var selectedAnswers = document.querySelectorAll("#answer-container input:checked");
        // Если ни один ответ не выбран, выводим сообщение об ошибке
        if (selectedAnswers.length === 0) {
          alert("Пожалуйста, выберите хотя бы один ответ.");
          return;
        }
        // Сохраняем выбранные ответы в массив
        var answers = [];
        for (var i = 0; i < selectedAnswers.length; i++) {
          answers.push(selectedAnswers[i].value);
        }
        // Очищаем контейнер с вариантами ответов
        document.getElementById("answer-container").innerHTML = "";
        // Если это был последний вопрос, выводим результаты
        if (currentQuestion === questions.length - 1) {
          showResults();
        } else {
          // Иначе выводим следующий вопрос
          currentQuestion++;
          showQuestion(currentQuestion);
        }
      });

      // Массив с вопросами
      var questions = [
        {
          text: "Какие виды деятельности вам нравятся?",
          options: [
            { text: "Реалистические", value: ""R" },
{ text: "Интеллектуальные", value: "I" },
{ text: "Художественные", value: "A" },

]
},
{

options: [
{ text: "Строительство и ремонт", value: "R" },

{ text: "Изобразительное искусство", value: "A" },
{ text: "Обслуживание и помощь людям", value: "S" },

{ text: "Компьютерные технологии", value: "C" }
]
},
{
text: "Какие навыки вы считаете наиболее важными?",
options: [
{ text: "Работа с инструментами и оборудованием", value: "R" },
{ text: "Логическое мышление и анализ", value: "I" },
{ text: "Художественный вкус и творческий потенциал", value: "A" },
{ text: "Умение общаться и помогать людям", value: "S" },
{ text: "Лидерские и организаторские навыки", value: "E" },
{ text: "Работа с компьютером и офисными программами", value: "C" }
]
}
];
  var currentQuestion = 0; // Текущий вопрос

  // Функция для отображения вопроса и вариантов ответов на странице
  function showQuestion(questionIndex) {
    var question = questions[questionIndex];
    // Выводим текст вопроса
    document.getElementById("question").textContent = question.text;
    // Создаем контейнер для вариантов ответов
    var answerContainer = document.getElementById("answer-container");
    // Добавляем варианты ответов в контейнер
    for (var i = 0; i < question.options.length; i++) {
      var option = question.options[i];
      var label = document.createElement("label");
      var checkbox = document.createElement("input");
      checkbox.type = "checkbox";
      checkbox.value = option.value;
      label.appendChild(checkbox);
      label.appendChild(document.createTextNode(option.text));
      answerContainer.appendChild(label);
    }
  }

  // Функция для вывода результатов теста
  function showResults() {
    // Считаем количество ответов каждого типа
    var counts = { R: 0, I: 0, A: 0, S: 0, E: 0, C: 0 };
    for (var i = 0; i < selectedOptions.length; i++) {
var option = selectedOptions[i];
counts[option]++;
}
    // Определяем наиболее подходящий тип личности по результатам теста
    var personalityType = "";
    var maxCount = 0;
    for (var type in counts) {
      if (counts[type] > maxCount) {
        personalityType = type;
        maxCount = counts[type];
      }
    }

    // Выводим результат на страницу
    document.getElementById("question").textContent = "Ваш тип личности: " + personalityType;
    document.getElementById("answer-container").innerHTML = "";
  }

  // Обработчик события для кнопки "Далее"
  document.getElementById("next-button").addEventListener("click", function() {
    // Получаем выбранные варианты ответов
    var selectedOptions = [];
    var checkboxes = document.getElementsByTagName("input");
    for (var i = 0; i < checkboxes.length; i++) {
      if (checkboxes[i].type === "checkbox" && checkboxes[i].checked) {
        selectedOptions.push(checkboxes[i].value);
      }
    }

    // Переходим к следующему вопросу, если он есть
    if (currentQuestion < questions.length - 1) {
      currentQuestion++;
      showQuestion(currentQuestion);
    } else { // Иначе выводим результаты теста
      showResults();
    }
  });

  // Начинаем тест с первого вопроса
  showQuestion(currentQuestion);
</script>
